# 初始化项目

  • mkdir project
  • cd projecct
  • npm init -y

# 安装 webpack

  • npm install -D webpack webpack-cli 默认是 webpack5, 最新版
  • npm install webpack@4 -D 安装 webpack4
  • webpack 中下载的东西都属于是开发依赖

# 创建入口文件

  • project/src/index.js

# 执行打包

  • 必须添加 mode

  • webpack ./src/index.js --output-path ./dist --mode=development

    • --mode 可以修改为 production, 是生产环境
    • --output-path 可以简写为 -o

  • 也可以只写一个 webpack, 这样也可以打包

# webpack 配置文件

const { resolve } = require('path')

module.exports = {
    // 打包模式, 分为 development 和 production
    mode: 'development',
    // 入口文件
    entry: './src/index.js',
    output: {
        // 输出目录, 输出目录必须为绝对路径, __dirname 是 nodejs 的一个变量, 表示的是当前文件所在目录的绝对路径
        path: resolve(__dirname, 'dist'),
        // 输出文件名
    	filename: 'main.js'
    },
        
    module: {
        rules: [
        	// 指定配置规则
	    ]
    },
	// 插件配置        
	plugins: [],

	// 开发服务器
	devServer: {}
}
