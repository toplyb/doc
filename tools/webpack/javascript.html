# js 语法检查

  • npm i eslint-loader eslint-config-airbnb-base eslint-plugin-import eslint -D

    const {resolve} = require('path');
const HtmlWebpackPlufin = require('html-webpack-plugin');

module.exports = {
    entry: './src/js/index.js',
    output: {
        filename: 'js/built.js',
        path: resolve(__dirname, 'build')
    },
    module: {
        rules: [
            {
                test: /\.less$/,
                use: [
                    'style-loader',
                    'css-loader',
                    'less-loader'
                ]
            },
            // js 语法检查
            {
                test: /\.js$/,
                exclude: /node_modules/,
                loader: 'eslint-loader',
                options: {
                    // 自动修改 eslint 错误
                    fix: true
                }
            }
        ]
    },
    plugins: [
        new HtmlWebpackPlugin({
            template: './src/index.html'
        })
    ],
    mode: 'development'
}

  • 想要实现 eslint 语法检查, 还需要在 package.json 文件中添加以下内容:

    "eslintConfig": {
    "extends": "airbnb-base"
}

# js 语法兼容性处理

# 方法1 (基础语法的兼容性处理)

  • npm i babel-loader @babel/core @babel/preset-env

    const {resolve} = require('path');

module.exports = {
    entry: './src/js/index.js',
    output: {
        filename: 'js/built.js',
        path: 'build'
    },
    module: {
        rules: [
            {
                test: /\.js$/,
                exclude: /node_modules/,
                options: [
                    // 预设, 指示 babel 应该做怎样的兼容性处理
                    presets: ['@babel/preset-env']
                ]
            }
        ]
    },
    plugins: [],
    mode: 'development'
}

# 方法2 (全部兼容性处理)

  • npm i @babel/polyfill -D

  • 在入口文件中, src/js/index.js 引入以下内容

    import '@babel/polyfill'

# 方法3 (按需加载)

  • npm i core-js -D

    const {resolve} = require('path');

module.exports = {
    entry: './src/js/index.js',
    output: {
        filename: 'js/built.js',
        path: 'build'
    },
    module: {
        rules: [
            {
                test: /\.js$/,
                exclude: /node_modules/,
                options: [
                    // 预设, 指示 babel 应该做怎样的兼容性处理
                    presets: [
                    	[
                    		'@babel/preset-env',
                    		{
                    			// 按需加载
                    			useBuiltIns: 'usage',
                    			// 指定 core-js 版本
                    			corejs: {
                    				version: 3
                    			}
                    		},
            				// 指定兼容性做到哪个版本的浏览器
            				targets: {
            					chrome: '60',
            					firefox: '60',
            					ie: '8',
            					safari: '10',
            					edge: '17'
            				}
		                ]
                	]
                ]
            }
        ]
    },
    plugins: [],
    mode: 'development'
}
