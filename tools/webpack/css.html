# 注意

  • 非 js 文件打包, 需要对应的 loader
  • css-loader 是将 css 转换为 js (将 css 输出到打包后的 js 文件中)
  • style-loader 是把包含 css 内容的 js 代码, 挂载到页面的 style 标签中

# 基本步骤

  • npm install -D css-loader style-loader
  • 在入口文件中使用 import 引入 css 文件
const { resolve } = require('path')

module.exports = {
    mode: 'development',
    entry: './src/index.js',
    output: {
        path: resolve(__dirname, 'dist'),
	    filename: 'main.js'
    },
        
	module: {
        rules: [
        	{
                // test 表示匹配哪些文件名
        		test: /\.css$/i,
                // 使用哪些 loader 来处理对应的文件, use 数组中的 loader 的执行顺序是从右到左, 从下到上执行的
        		use: [
                    // 创建 style 标签, 将 js 中的样式资源插入到其中, 然后添加到 HTML 中的 header 中生效
        			'style-loader',
                    // css-loader 是将 css 文件变成 commonjs 模块加载到 js 中, 里面内容是样式字符串
        			'css-loader'
			    ]
        	}
    	]
    },
        
    plugins: [],
	devServer: {}
}

# 打包 less

  • npm install -D less less-loader

  • 在入口文件中引入 less 文件

  • Webpack 中的配置如下:

    const { resolve } = require('path')

module.exports = {
    mode: 'development',
    entry: './src/index.js',
    output: {
        path: resolve(__dirname, 'dist'),
    	filename: 'main.js'
    },
	module: {
        rules: [
            {
                test: /\.less$/i,
                use: [
                    'style-loader',
                    'css-loader',
                    // 将 less-loader 编译成 css 文件
                    'less-loader'
                ]
            }
        ]
    }
}

# 将 css 打包成独立的文件

  • npm install -D mini-css-extract-plugin

  • Webpack 配置

    const { resolve } = require('path');
const MiniCssExtractPlugin = require('mini-css-extract-plugin');

module.exports = {
    mode: 'development',
    entry: './src/index.js',
    output: {
        path: resolve(__dirname, 'dist'),
    	filename: 'main.js'
    },
	modules: {
        rules: [
            {
                test: /\.css$/i
                use: [
                	// 将 css 打包到单独的文件
                	MiniCssExtractPlugin.loader,
                	'css-loader'
                ]
            }
        ]
    },
	plugins: [
        // 这样写的话默认是放在 dist 目录中的 main.css
        // new MiniCssExtractPlugin()
        // 指定文件路径和文件名
		new MiniCssExtractPlugin({
            // 放在 /dist/css 目录中, 使用文件本来的名字
            filename: 'css/[name].css'
        })
    ]
}

# 解决 css 兼容性问题

  • npm install -D postcss-loader autoprefixer

  • webpack 配置

    const { resolve } = require('path')
const MiniCssExtractPlugin = require('mini-css-extract-plugin')

module.exports = {
    mode: 'development',
    entry: './src/index.js',
    output: {
        path: resolve(__dirname, 'dist'),
        filename: 'main.js'
    },
    module: {
        rules: [
            {
                test: /\.css$/i,
                use: [
                    MiniCssExtractPlugin.loader,
                    'css-loader',
                    'postcss-loader'
                ]
            }
        ]
    }
}

  • 新建 postcss.config.js 文件并写入

    module.exports = {
    'plugins': [
        require('autoprefixer')
    ]
}

  • 在 package.json 文件中加入:

    ...
"borwserslist": [
    // 兼容最新版本
    "last 1 version",
    // 代表全球超过 1% 使用的浏览器
    "> 1%",
    // 覆盖 99.5% 的浏览器
    "cover 99.5%"
]

# 生产环境提取 css 成单独文件

  • npm i mini-css-extract-plugin -D

    const {resolve} = require('path');
const HtmlWebpackPlugin = require('html-webpack-plugin');
const MiniCssWebpackPlugin = require('mini-css-extract-plugin');

module.exports = {
    entry: './src/js/index.js',
    output: {
        filename: 'js/built.js',
        path: resolve(__dirname, 'build')
    },
    module: {
        rules: [
            {
                test: /\.less$/,
                use: [
                    MiniWebpackPlugin.loader,
                    'css-loader',
                    'less-loader'
                ]
            }
        ]
    },
    plugins: [
        new HtmlWebpackPlugin({
            template: './src/index.html'
        }),
        new MiniCssWebpackPlugin({
            filename:' css/buily.css'
        })
    ]
}

# 处理 css 浏览器兼容性

  • npm i postcss-loader postcss-preset-env -D

    const {resolve} = require('psth');
const HtmlWebpackPlugin = require('html-webpack-plugin');
const MiniCssWebpackPlugin = reqiure('mini-css-webpack-plugin');

module.export = {
    entry: './src/js/index.js',
    output: {
        filename: 'js/built.js',
        path: resolve(__dirname, 'build')
    },
    module: {
        rules: [
            {
                test: /\.less$/,
                use: [
                    MiniCssWebpackPlugin.loader,
                    'css-loader',
                    {
                        loader: 'postcss-loader',
                        options: {
                            postcssOptions: {
                                plugins: [
                                    [
                                        'postcss-preset-env'
                                    ]
                                ]
                            }
                        }
                    },
                    'less-loader'
                ]
            }
        ]
    },
    plugins: [
        new HtmlWebpackPlugin({
            template: './src/index.html'
        }),
        new MiniCssWebpackPlugin({
            filename: 'css/built.js'
        })
    ],
    mode: 'development'
}

  • 想要处理 css 浏览器兼容问题, 还需要在 package.json 文件中添加以下内容

    "borswerList":{
    // 开发环境
    "development": [
        // 兼容最新版本的 chrome 浏览器
        "last 1 chrome version"
    ],
    // 生成环境
    "production": [
        // 兼容百分之九十八以上的浏览器
        ">0.2%",
        // 不兼容已经死了的
        "not dead",
        // 不兼容 op_mini 的
        "not op_mini all"
    ]
}

# 压缩 css

  • npm i optimize-css-assets-webpack-plugin -D

    const {resolve} = require('path');
const HtmlWebpackPlugin = require('html-webpack-plugin');
const MiniCssWebpackPlugin = require('mini-css-webpack-plugin');
const OptimizeCssAssetsWebpackPlugin = require('optimize-css-assets-webpack-plugin');

module.export = {
    entry: './src/js/index.js',
    output: {
        filename: 'js/built.js',
        path: resolve(__dirname, 'build')
    },
    module: {
        rules: [
            {
                test: /\.less$/,
                use: [
                    MiniCssWebpackPlugin.loader,
                    'css-loader',
                    {
                        loader: 'postcss-loader',
                        options: {
                            postcssOptions: {
                                plugins: [
                                    [
                                        'postcss-preset-env'
                                    ]
                                ]
                            }
                        }
                    },
                    'less-loader'
                ]
            }
        ]
    },
    plugins: [
        new HtmlWebpackPlugin({
            template: './src/index.html'
        }),
        new MiniCssWebpackPlugin({
            filename: 'css/built.css'
        }),
        new OptimizeCssAssetsWebpackPlugin()
    ],
    mode:'development'
}
