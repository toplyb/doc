# 写法一

<script>
const temp = {
  template: `<div>hello world</div>`,
};

const app = Vue.createApp(temp);
app.mount("#app");
</script>

<script>
Vue.createApp({
  template: `<h2>hello world</h2>`,
}).mount("#app");
</script>

# 写法二

<script type="x-template" id="ceshi">
<h2>hello world</h2>
</script>

<script>
Vue.createApp({
  template: "#ceshi",
}).mount("#app");
</script>

# 写法三

<template id="ceshi">
  <h2>hello world</h2>
</template>

<script>
Vue.createApp({
  template: "#ceshi",
}).mount("#app");
</script>
