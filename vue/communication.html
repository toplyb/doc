# 子传父

Children.vue

<div>
    {{qian}}
</div>
<button @click="changeMoney">
    
</button>
<script>
import { ref } from "vue";
// 子组件接收父组件的数据使用 props 即可
export default {
  name: "app",
  props: {
    qian: {
      type: Number,
      default: 0,
    },
  },
  // ctx 全程是 context, 其中有一个 emit, 会触发自定义函数
  setup(props, ctx) {
    // 获取父组件中的数据
    console.log(props.qian);

    const changeMoney = () => {
      ctx.emit("change-money", 50);
    };

    return { changeMoney };
  },
};
</script>

# 多级组件通信

  • 父组件中有很多子组件, 孙子组件等后代组件, 这些组件要共享父组件的数据, 需要用到 provide 和 inject

    Father.vue

    <Son />
<script>
import { provide, ref } from "vue";
import Son from "./Son";
export default {
  name: "app",
  components: {
    Son,
  },
  setup() {
    const money = ref(100);

    // 在父组件中定义修改数据的函数
    const changeMoney = (m) => {
      money.value = money.value - m;
    };
    // 将数据提供给后代组件 provide, 如果父组件中数据变化, 后代组件中的数据也会跟着变化
    provide("money", money);

    // 将修改数据的函数传递给后代组件
    provide("change", changeMoney);
    return { money };
  },
};
</script>

    Son.vue

    <button @click="updateMoney">
    改变数据
</button>
<script>
import { inject } from "vue";
export default {
  name: "app",
  setup() {
    // 接收祖先组件提供的数据, 在孙子节点中也是这样获取, 此时在后代组件中是不能修改数据的, 一般是谁定义, 谁修改, 遵循单向数据流
    // 如果一定要在后代组件中修改数据的话, 需要通知最顶层的父组件
    const money = inject("money");

    // 接受父组件中传递过来的修改数据的函数
    const change = inject("change");

    const updateMoney = () => {
      change(20);
    };
    return { money, updateMoney };
  },
};
</script>
