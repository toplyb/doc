const axisData = [];

for (let i = 0; i < data.length; i++) {
  const height = data[i]["height"];
  const weight = data[i]["weight"];
  const temp = [height, weight];
  axisData.push(temp);
}

const options = {
  xAxis: {
    type: "value",
    scale: true,
  },
  yAxis: {
    type: "value",
    scale: true,
  },
  series: [
    {
      // type: 'scatter', // 设置散点图
      type: "effectScatter", // 设置涟漪动画
      showEffectOn: "render", // render 表示每个散点都有动画, emphasis 表示鼠标移入后才有动画
      // 设置涟漪的范围
      rippleEffect: {
        scale: 6,
      },
      data: axisData,
      // 设置散点的大小
      // symbolSize: 20
      symbolSize: function(arg) {
        const height = arg[0] / 100;
        const weight = arg[1];
        // bmi = 体重(kg) / (身高m * 身高), 大于 28, 表示肥胖
        const bmi = weight / (height * height);
        if (bmi > 28) {
          return 20;
        }
        return 5;
      },
      // 设置气泡的颜色
      itemStyle: {
        color: function(arg) {
          const height = arg["data"][0] / 100;
          const weight = arg["data"][1];
          const bmi = weight / (height * height);
          if (bmi > 28) {
            return "red";
          }
          return "green";
        },
      },
    },
  ],
};

myEchart.setOption(options);
