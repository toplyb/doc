# 线性渐变

const myEchart = echarts.init(document.querySelector("div"));

const xDataArr = [
  "张三",
  "李四",
  "王五",
  "闰土",
  "小明",
  "茅台",
  "二妞",
  "大强",
];
const yDataArr = [88, 92, 63, 77, 94, 80, 72, 86];

const options = {
  title: {
    text: "成绩",
  },
  xAxis: {
    type: "category",
    data: xDataArr,
  },
  yAxis: {
    type: "value",
  },
  series: [
    {
      name: "语文",
      type: "bar",
      data: yDataArr,
      itemStyle: {
        color: {
          // 线性渐变
          type: "linear",
          x: 0,
          y: 0,
          x2: 0,
          y2: 1,
          colorStops: [
            {
              // 0% 的时候是红色
              offset: 0,
              color: "red",
            },
            {
              // 100% 的时候是红色
              offset: 1,
              color: "blue",
            },
          ],
        },
      },
    },
  ],
};

myEchart.setOption(options);

# 径向渐变

const myEchart = echarts.init(document.querySelector("div"));

const xDataArr = [
  "张三",
  "李四",
  "王五",
  "闰土",
  "小明",
  "茅台",
  "二妞",
  "大强",
];
const yDataArr = [88, 92, 63, 77, 94, 80, 72, 86];

const options = {
  title: {
    text: "成绩",
  },
  xAxis: {
    type: "category",
    data: xDataArr,
  },
  yAxis: {
    type: "value",
  },
  series: [
    {
      name: "语文",
      type: "bar",
      data: yDataArr,
      itemStyle: {
        // 径向渐变
        color: {
          type: "radial",
          // x, y 是圆心点的位置
          x: 0.5,
          y: 0.5,
          // r 是半径
          r: 0.5,
          colorStops: [
            {
              // 0% 的时候是红色
              offset: 0,
              color: "red",
            },
            {
              // 100% 的时候是红色
              offset: 1,
              color: "blue",
            },
          ],
        },
      },
    },
  ],
};

myEchart.setOption(options);
