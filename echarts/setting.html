# 标题

const myEchart = echarts.init(document.querySelector("div"));

const options = {
  title: {
    text: "标题", // 标题文本
    link: "http://www.baidu.com", // 超链接
    textStyle: {
      // 设置颜色, 还可以设置其他样式, 可以查看文档
      color: "skyblue",
    },
    borderWidth: 5,
    borderColor: "purple",
    borderRadius: 5,
    // 设置标题的位置
    left: 50,
  }, // 主标题
  // 配置 x 轴
  xAxis: {
    type: "category", // 类目轴
    data: ["小明", "小红", "小王"],
  },
  // 配置 y 轴
  yAxis: {
    type: "value", // 数值轴, 值是 series 中的 data
  },
  series: [
    {
      name: "语文",
      type: "bar", // bar 表示的是柱状图, line 表示的折线图 ...
      data: [70, 80, 90],
    },
  ],
};

myEchart.setOption(options);

# tooltip 提示框组件

  • 提示框组件, 用于配置鼠标滑过或点击图表时的显示框
const myEchart = echarts.init(document.querySelector("div"));

const xDataArr = [
  "张三",
  "李四",
  "王五",
  "闰土",
  "小明",
  "茅台",
  "二妞",
  "大强",
];
const yDataArr = [88, 92, 63, 77, 94, 80, 72, 86];

const options = {
  xAxis: {
    type: "category",
    data: xDataArr,
  },
  yAxis: {
    type: "value",
  },
  series: [
    {
      name: "语文",
      type: "bar",
      data: yDataArr,
    },
  ],
  tooltip: {
    // 设置显示提示框的触发的类型
    trigger: "item",
    // 设置显示提示框的触发的时机
    triggerOn: "click",
    // 设置提示框中显示的内容
    formatter: "haha",
  },
};

myEchart.setOption(options);

# legend 图例

  • 用于筛选系列, 需要和 series 配合使用
const myEchart = echarts.init(document.querySelector("div"));

const xDataArr = [
  "张三",
  "李四",
  "王五",
  "闰土",
  "小明",
  "茅台",
  "二妞",
  "大强",
];
const yDataArr = [88, 92, 63, 77, 94, 80, 72, 86];
const yDataArr2 = [93, 60, 61, 62, 85, 79, 92, 30];

const options = {
  xAxis: {
    type: "category",
    data: xDataArr,
  },
  yAxis: {
    type: "value",
  },
  series: [
    {
      name: "语文",
      type: "bar",
      data: yDataArr,
    },
    {
      name: "数学",
      type: "bar",
      data: yDataArr2,
    },
  ],
  legend: {
    // 需要和 series 中的 name 值一样
    data: ["语文", "数学"],
  },
};

myEchart.setOption(options);

# 直角坐标系常用配置

const myEchart = echarts.init(document.querySelector("div"));

const xDataArr = [
  "张三",
  "李四",
  "王五",
  "闰土",
  "小明",
  "茅台",
  "二妞",
  "大强",
];
const yDataArr = [88, 92, 63, 77, 94, 80, 72, 86];

const options = {
  // grid 用来设置直角坐标系的布局和大小
  grid: {
    // 显示边框
    show: true,
    // 设置边框的粗细
    borderWidth: 10,
    // 设置边框的颜色
    borderColor: "red",
    top: 100,
    width: 500,
  },
  xAxis: {
    type: "category",
    data: xDataArr,
  },
  yAxis: {
    type: "value",
  },
  series: [
    {
      name: "语文",
      type: "bar",
      data: yDataArr,
    },
  ],
  // 区域缩放
  dataZoom: [
    {
      // slider 是滑块, inside 是依靠鼠标滚轮或双指缩放
      type: "inside",
      // 指明产生作用的轴, 一般写 0
      xAxisIndex: 0,
    },
    {
      type: "inside",
      yAxisIndex: 0,
      // 指明初始状态的缩放情况
      start: 0,
      end: 10,
    },
  ],
};

myEchart.setOption(options);

# 增量动画

<button id="updateData">修改数据</button>
<button id="addData">新增数据</button>

<script>
  const myEchart = echarts.init(document.querySelector("div"));

  let xDataArr = [
    "1月",
    "2月",
    "3月",
    "4月",
    "5月",
    "6月",
    "7月",
    "8月",
    "9月",
    "10月",
    "11月",
    "12月",
  ];
  let yDataArr = [
    3000,
    2800,
    900,
    1000,
    800,
    700,
    1400,
    1300,
    900,
    1000,
    800,
    600,
  ];

  const options = {
    xAxis: {
      type: "category",
      data: xDataArr,
      boundaryGap: false,
    },
    yAxis: {
      type: "value",
      scale: true,
    },
    series: [
      {
        name: "康师傅",
        type: "line",
        data: yDataArr,
      },
    ],
  };

  myEchart.setOption(options);

  const updateBtn = document.getElementById("updateData");
  updateBtn.onclick = function() {
    const newYDataArr = [
      4000,
      2000,
      1000,
      800,
      1200,
      500,
      1400,
      1000,
      600,
      1200,
      980,
      1060,
    ];
    // 新旧 options 是相互叠加的,不是覆盖的
    const options = {
      series: [
        {
          data: newYDataArr,
        },
      ],
    };

    myEchart.setOption(options);
  };

  const addBtn = document.getElementById("addData");
  addBtn.onclick = function() {
    xDataArr.push("13月");
    yDataArr.push(1400);

    const options = {
      xAxis: {
        data: xDataArr,
      },
      series: [
        {
          data: yDataArr,
        },
      ],
    };

    myEchart.setOption(options);
  };
</script>

    # 动画配置

    const myEchart = echarts.init(document.querySelector("div"));

const xDataArr = [
  "张三",
  "李四",
  "王五",
  "闰土",
  "小明",
  "茅台",
  "二妞",
  "大强",
];
const yDataArr = [88, 92, 63, 77, 94, 80, 72, 86];

const options = {
  // 开启动画, 默认为 true
  animation: true,
  // 设置动画时长, 单位是毫秒, 也可以设置回调函数
  // animationDuration: 3000,
  animationDuration: function(arg) {
    // arg 是下标, 代码着图中每一个数值
    return 2000 * arg;
  },
  // 设置缓动动画的模式, 先快后慢还是匀速之类的
  animationEasing: "linear",
  // 设置动画阈值, 当同一类型的数据超过这个值了之后, 就不会显示动画了
  animationThreshold: 3,
  title: {
    text: "成绩",
  },
  xAxis: {
    type: "category",
    data: xDataArr,
  },
  yAxis: {
    type: "value",
  },
  series: [
    {
      name: "语文",
      type: "bar",
      data: yDataArr,
    },
  ],
};

myEchart.setOption(options);
