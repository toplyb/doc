# 基础地图

const myEchart = echarts.init(document.querySelector("div"));

// 需要加载地图的数据
$.get("json/map/china.json", function(res) {
  echarts.registerMap("chinaMap", res);

  const options = {
    geo: {
      type: "map",
      map: "chinaMap",
      // 可以缩放, 拖拽地图
      roam: true,
      label: {
        // 名称显示
        show: true,
      },
      // 设置初始化缩放比例
      zoom: 2,
      // 设置地图的中心点, 值是中心点的坐标
      // center: []
    },
  };

  myEchart.setOption(options);
});

# 某个省份地图

const myEchart = echarts.init(document.querySelector("div"));

$.get("json/map/anhui.json", function(res) {
  echarts.registerMap("anhui", res);

  const options = {
    geo: {
      type: "map",
      map: "anhui",
      roam: true,
      label: {
        show: true,
      },
      zoom: 1.2,
      center: [116.507676, 31.752889],
    },
  };

  myEchart.setOption(options);
});

$.get("json/map/china.json", function(res) {
  echarts.registerMap("chinaMap", res);

  const options = {
    geo: {
      type: "map",
      map: "chinaMap",
      // 可以缩放, 拖拽地图
      roam: true,
      label: {
        // 名称显示
        show: true,
      },
      // 设置初始化缩放比例
      zoom: 2,
      // 设置地图的中心点, 值是中心点的坐标
      // center: []
    },
    series: [
      {
        data: airData,
        // 将空气质量的数据和第 0 个 geo 配置关联在一起
        geoIndex: 0,
        type: "map",
      },
      {
        // 配置散点数据
        data: scatterData,
        type: "effectScatter",
        // 指明散点使用的坐标系统是 geo 的坐标系统
        coordinateSystem: "geo",
        rippleEffect: {
          // 设置涟漪动画的涟漪效果
          scale: 10,
        },
      },
    ],
    visualMap: {
      min: 0,
      max: 300,
      inRange: {
        // 颜色由白色慢慢过渡到红色
        color: ["white", "red"],
      },
      // 出现滑块, 对数据进行一个筛选
      calculable: true,
    },
  };

  myEchart.setOption(options);
});
