# 雷达图

const myEchart = echarts.init(document.querySelector("div"));

// 各个维度的最大值
const dataMax = [
  {
    name: "易用性",
    max: 100,
  },
  {
    name: "功能",
    max: 100,
  },
  {
    name: "拍照",
    max: 100,
  },
  {
    name: "跑分",
    max: 100,
  },
  {
    name: "续航",
    max: 100,
  },
];

const options = {
  radar: {
    // 配置各个维度的最大值
    indicator: dataMax,
    // 配置雷达图最外层的图形样式, 默认是 polygon
    shape: "circle",
  },
  series: [
    {
      data: [
        {
          name: "华为手机1",
          value: [80, 90, 80, 82, 90],
        },
        {
          name: "中兴手机1",
          value: [70, 82, 75, 70, 78],
        },
      ],
      type: "radar",
      label: {
        // 显示数值
        // show: true
      },
      // 将每一个产品的雷达图形成阴影的面积
      areaStyle: {},
    },
  ],
};

myEchart.setOption(options);
