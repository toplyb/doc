# On/Off

const myEchart = echarts.init(document.querySelector("div"));

const data = [
  {
    name: "淘宝",
    value: 11231,
  },
  {
    name: "京东",
    value: 22673,
  },
  {
    name: "唯品会",
    value: 6123,
  },
  {
    name: "1号店",
    value: 8989,
  },
  {
    name: "聚美优品",
    value: 6700,
  },
];

const options = {
  // 添加图例
  legend: {
    data: ["淘宝", "京东", "唯品会", "1号店", "聚美优品"],
  },
  series: [
    {
      type: "pie",
      data: data,
    },
  ],
};

myEchart.setOption(options);
// 监听常见鼠标的事件, 如点击, 移入移出等
myEchart.on("click", (arg) => {
  // arg 表示点击对象的基本信息
  console.log(arg);
});

// 解绑点击事件
myEchart.off("click");

// 也可以监听 ECharts 事件, 常用事件可以参考文档 https://echarts.apache.org/zh/api.html#events
// 监听图例的点击事件
myEchart.on("legendselectchanged", (arg) => {
  // arg 表示点击的图例的对象
  console.log(arg);
});

# dispatchAction

<button id="btn" onclick="btn1()">高亮</button>
<script>
  const myEchart = echarts.init(document.querySelector("div"));

  const data = [
    {
      name: "淘宝",
      value: 11231,
    },
    {
      name: "京东",
      value: 22673,
    },
    {
      name: "唯品会",
      value: 6123,
    },
    {
      name: "1号店",
      value: 8989,
    },
    {
      name: "聚美优品",
      value: 6700,
    },
  ];

  const options = {
    // 添加图例
    legend: {
      data: ["淘宝", "京东", "唯品会", "1号店", "聚美优品"],
    },
    series: [
      {
        type: "pie",
        data: data,
      },
    ],
    tooltip: {
      // 设置显示提示框的触发的类型
      trigger: "item",
      // 设置显示提示框的触发的时机
      triggerOn: "click",
    },
  };

  // dispatchAction 模拟用户行为
  function btn1() {
    myEchart.dispatchAction({
      // 事件类型
      type: "highlight",
      // 图表索引, 即 options 中的 series 数组中的第几项
      seriesIndex: 0,
      // 图表中哪一项高亮
      dataIndex: 1,
    });

    myEchart.dispatchAction({
      type: "showTip",
      seriesIndex: 0,
      dataIndex: 1,
    });
  }
</script>

# clear

// 清空当前实例, 会移除实例中所有的组件和图表
// 清空之后可以再 setOption
myEcharts.clear();

# dispose

// 销毁 ECharts, 销毁之后再次 setOption 的话无效
myEcharts.dispose();
