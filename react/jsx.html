# 嵌入 JS 表达式

  • 语法: { JS表达式 }
const name = 'xiaosheng'
const info = (<div>我叫{name}</div>)

# 条件渲染

const loading = true
const loadData = () => {
  if (loading) {
    return <div>loading...</div>
  }
  return <div>请求结束</div>
}

const data = (
  <h1>
    条件渲染
    {loadData()}
  </h1>
)

# 列表渲染

const languages = [
  {
    id: 1,
    name: 'html'
  },
  {
    id: 2,
    name: 'css'
  },
  {
    id: 3,
    name: 'javascript'
  }
]

const data = (
  <ul>
    {languages.map(item => {
      return <li key={item.id}>{item.name}</li>
    })}
  </ul>
)
