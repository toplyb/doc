# 基本使用

  • / 会匹配到所有的路由
  • /first 会匹配到 first, 但是 /first/a, /first/a/b... 匹配不到
  • 只有使用 BrowserRouter 组件包围的内容, 才能使用路由, 一般来说, 一个应用只需要使用一次
  • BrowserRouter 表示非 hash 路由, HashRouter 表示 hash 路由
  • Route 组件需要使用 Routes 包围
  • Route 组件写在哪里, 渲染出来的组件就显示在哪里, 其中 path 属性是路由规则, element 表示要渲染的组件
import {BrowserRouter as Router, Routes, Route, Link} from 'react-router-dom';

class First extends React.Component {
  render() {
    return (
      <p>这是页面一的内容</p>
    )
  }
}

const App = () => (
  <Router>
    <div>
      <h1>路由</h1>
      <Link to='/first'>页面一</Link>
      <Routes>
        <Route path='/first' element={<First></First>}></Route>
      </Routes>
    </div>
  </Router>
)
ReactDOM.render(<App></App>, document.getElementById('root'))

import {useNavigate} from 'react-router-dom'

function Login() {

  const navigate = useNavigate()

  const about = () => {
    navigate('/about')
  }

  return (
    <div>
      login
      <button onClick={about}>跳转到关于页面</button>
    </div>
  )
}

export default Login

# 路由传参

  • searchParams 传参: 使用 useSearchParams 函数来获取到参数
// Login.js
import {useNavigate} from 'react-router-dom'

function Login() {

  const navigate = useNavigate()

  const about = () => {
    // 传递参数
    navigate('/about?id=100')
  }

  return (
    <div>
      login
      <button onClick={about}>about</button>
    </div>
  )
}

export default Login

// About.js
import {useSearchParams} from "react-router-dom";

function About() {

  const [params] = useSearchParams()
  console.log(params.get('id'));

  return (
    <div>about</div>
  )
}

export default About

# 二级路由

  • 在 Route 中嵌套Route 即可, 同时需要在主页面将添加 <Outlet></Outlet> 来表示二级页面
// App.js
import Layout from "./Layout";
import Article from "./Article";
import Login from "./Login";
import {BrowserRouter, Routes, Route} from "react-router-dom";

function App() {
  return (
    <BrowserRouter>
      <Routes>
        <Route path='/' element={<Layout></Layout>}>
          {/*定义二级路由*/}
          <Route path='/article' element={<Article></Article>}></Route>
        </Route>
        <Route path='/login' element={<Login></Login>}></Route>
      </Routes>
    </BrowserRouter>
  )
}

export default App

// Layout.js
import {Outlet} from 'react-router-dom'

function Layout() {
  return (
    <div>
      layout
      <Outlet></Outlet>
    </div>
  )
}

export default Layout
  • 默认路由: 将路由中的 path 改为 index 即可
import Layout from "./Layout";
import Article from "./Article";
import Login from "./Login";
import {BrowserRouter, Routes, Route, Link} from "react-router-dom";

function App() {
  return (
    <BrowserRouter>
      <Routes>
        <Route path='/' element={<Layout></Layout>}>
          {/*定义二级路由*/}
          <Route index element={<Article></Article>}></Route>
        </Route>
        <Route path='/login' element={<Login></Login>}></Route>
      </Routes>
    </BrowserRouter>
  )
}

export default App
