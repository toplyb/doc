# 优势

  1. 简单: 可以编写没有模板的原生 js 代码
  2. 轻松实现最优渲染: 依赖自动追踪最小渲染优化
  3. 架构自由: 可移植, 可测试

# 安装

  • npm i mobx mobx-react-lite
  • mobx-react-lite 这个包只能在函数组件中使用, 如果想要在类组件中使用, 则需要安装 mobx-react

# 使用方法

# 数字自增案例

  • store/counter.js
import {makeAutoObservable} from "mobx";

class CounterStore {
  // 定义数据
  count = 0

  constructor() {
    // 将数据设置成响应式
    makeAutoObservable(this)
  }

  // 定义 action 函数, 来修改数据
  changeCount = () => {
    this.count++
  }
}

const counterStore = new CounterStore()
export {counterStore}
  • App.js
import {counterStore} from "./store/counter";
import {observer} from "mobx-react-lite";

function App() {
  return (
    <div className="App">
      app
      {counterStore.count}
      <button onClick={counterStore.changeCount}>+1</button>
    </div>
  );
}

// 使用 observer 将 react 和 mobx 联系起来
export default observer(App);

# 计算属性

  • counter.js
import {makeAutoObservable} from "mobx";

class CounterStore {
  list = [1,2,3,4,5]

  constructor() {
    // 将数据设置成响应式
    makeAutoObservable(this)
  }

  // get 是修饰符
  get filterList() {
    return this.list.filter(item => item > 2)
  }

  updateList = () => {
    this.list.push(6,7,8)
  }
}

const counterStore = new CounterStore()
export {counterStore}
  • App.js
import {counterStore} from "./store/counter";
import {observer} from "mobx-react-lite";

function App() {
  return (
    <div className="App">
      app
      {counterStore.filterList.join(',')}
      <button onClick={counterStore.updateList}>更新</button>
    </div>
  );
}

export default observer(App);
